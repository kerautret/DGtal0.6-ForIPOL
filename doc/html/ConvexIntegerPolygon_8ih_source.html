256 inline
257 typename DGtal::ConvexIntegerPolygon<TSpace,TSequence>::SizeCouple
258 DGtal::ConvexIntegerPolygon<TSpace,TSequence>::
259 findCut( Iterator & it_next_is_outside, Iterator & it_next_is_inside,
260  const HalfSpace & hs )
261 {
262  Size nbWithin = NumberTraits<Size>::ZERO;
263  Size nb = NumberTraits<Size>::ZERO;
264  Iterator it = begin();
265  Iterator it_prev = it;
266  Iterator it_end = end();
267  it_next_is_outside = it_next_is_inside = it_end;
268  if ( it == it_end )
269  return std::make_pair( NumberTraits<Size>::ZERO, NumberTraits<Size>::ZERO );
270  bool visibility_begin_vtx; // visibility of begin vertex.
271  bool visibility_prev_vtx; // visibility of previous vertex when visiting the list.
272  bool visibility_vtx; // visibility of current vertex when visiting the list.
273  ++nb;
274  if ( ( visibility_begin_vtx = hs( *it++ ) ) ) ++nbWithin; // Assignment
275  visibility_prev_vtx = visibility_begin_vtx;
276  for ( ; it != it_end; ++it, ++nb )
277  {
278  if ( ( visibility_vtx = hs( *it ) ) ) ++nbWithin; // Assignment
279  if ( visibility_vtx != visibility_prev_vtx )
280  {
281  if ( visibility_prev_vtx ) it_next_is_outside = it_prev;
282  else it_next_is_inside = it_prev;
283  }
284  visibility_prev_vtx = visibility_vtx;
285  it_prev = it;
286  }
287  if ( visibility_vtx != visibility_begin_vtx )
288  {
289  if ( visibility_vtx ) it_next_is_outside = it_prev;
290  else it_next_is_inside = it_prev;
291  }
292  ASSERT( ( nbWithin == 0 ) || ( nbWithin == size() )
293  || ( ( it_next_is_outside != it_end ) && ( it_next_is_inside != it_end )
294  && ( it_next_is_inside != it_next_is_outside ) ) );
295  return std::make_pair( nbWithin, nb );
296 }

299 template <typename TSpace, typename TSequence>
300 inline
301 bool
302 DGtal::ConvexIntegerPolygon<TSpace,TSequence>::
303 cut( const HalfSpace & hs )
304 {
305  Iterator it_next_is_outside;
306  Iterator it_next_is_inside;
307  SizeCouple nbs = findCut( it_next_is_outside, it_next_is_inside, hs );
308 
309  // Take care of easy cases.
310  if ( nbs.first == nbs.second ) { return false; }
311  if ( nbs.first == NumberTraits<Size>::ZERO ) { clear(); return true; }
312 
313  // Otherwise, determines A1B1 and A2B2.
314  twiceArea(); // result in _a;
315  HalfSpace hs1 = halfSpace( it_next_is_outside );
316  HalfSpace hs3 = halfSpace( it_next_is_inside );
317  //hs3.negate();
318  _A1 = *it_next_is_outside;
319  ++it_next_is_outside;
320  if ( it_next_is_outside == end() ) it_next_is_outside = begin();
321  _B1 = *it_next_is_outside;
322  _B2 = *it_next_is_inside;
323  ++it_next_is_inside;
324  if ( it_next_is_inside == end() ) it_next_is_inside = begin();
325  _A2 = *it_next_is_inside;
326 
327  // Erases outside vertices.
328  while ( it_next_is_outside != it_next_is_inside )
329  {
330  it_next_is_outside = erase( it_next_is_outside );
331  if ( it_next_is_outside == end() ) it_next_is_outside = begin();
332  }
333  // Both iterators point on the right place.
334  if ( _a > NumberTraits<Integer>::ZERO )
335  { //convex not reduced to a straight line segment
336  std::insert_iterator<Base> itOut
337  = std::inserter<Base,Iterator>( *this, it_next_is_outside );
338  computeConvexHullBorder( itOut, _A1, _A2, hs1, hs, hs3 );
339  }
340  else //convex reduced to a straight line segment
341  {
342  //compute the new extremity of the straight line segment
343  _v = _B1 - _A1;
344  _ic.reduce( _v );
345  _a = ( hs.c - hs.N.dot( _A1 ) ) / ( hs.N.dot( _v ) );
346  _A1 += _v * _a;
347  insertBefore( it_next_is_outside, _A1 );
348  }
349  purge(); // O(n)
350  return true;
351 }

354 inline
355 typename DGtal::ConvexIntegerPolygon<TSpace,TSequence>::HalfSpace
356 DGtal::ConvexIntegerPolygon<TSpace,TSequence>::
357 halfSpace( ConstIterator it ) const
358 {
359  Point A( *it ); ++it;
360  if ( it == end() ) it = begin();
361  Point B( *it ); ++it;
362  if ( it == end() ) it = begin();
363  _N[ 0 ] = A[ 1 ] - B[ 1 ];
364  _N[ 1 ] = B[ 0 ] - A[ 0 ];
365  _ic.getDotProduct( _c, _N, A );
366  _ic.getDotProduct( _c1, _N, *it );
367  if ( _c1 > _c )
368  {
369  _N.negate();
370  _c = -_c;
371  }
372  //simplification of the constraint
373  _ic.getGcd( _g, _N[ 0 ], _N[ 1 ] );
374  _N /= _g;
375  return HalfSpace( _N, _ic.floorDiv( _c, _g) );
376 }

409 template <typename TSpace, typename TSequence>
410 template <typename DigitalSet>
411 void
412 DGtal::ConvexIntegerPolygon<TSpace,TSequence>::
413 getIncludedDigitalPoints( DigitalSet & aSet ) const
414 {
415  BOOST_CONCEPT_ASSERT(( CDigitalSet< DigitalSet > ));
416  typedef typename DigitalSet::Domain DigitalSetDomain;
417  BOOST_STATIC_ASSERT(( ConceptUtils::SameType< DigitalSetDomain, Domain >::value ));
418  typedef typename DigitalSetDomain::ConstIterator DomainConstIterator;
419  typedef typename DigitalSet::Iterator DigitalSetIterator;
420  aSet.clear();
421  // case 1 : empty polygon.
422  if ( empty() ) return;
423  // case 2 : one vertex.
424  size_t s = size();
425  if ( s == 1 )
426  {
427  aSet.insert( *begin() );
428  return;
429  }
430  // case 3 : 2 vertices
431  if ( s == 2 )
432  {
433  ConstIterator it_vtx = begin();
434  ConstIterator it_vtx2 = it_vtx; ++it_vtx2;
435  HalfSpace hs = halfSpace( it_vtx );
436  Vector orthN( -hs.N[ 1 ], hs.N[ 0 ] );
437  HalfSpace hs_orth( orthN, _ic.dotProduct( orthN, *it_vtx ) );
438  if ( ! hs_orth( *it_vtx2 ) ) hs_orth.negate();
439  HalfSpace hs2 = halfSpace( it_vtx2 );
440  Vector orthN2( -hs2.N[ 1 ], hs2.N[ 0 ] );
441  HalfSpace hs2_orth( orthN2, _ic.dotProduct( orthN2, *it_vtx2 ) );
442  if ( ! hs2_orth( *it_vtx ) ) hs2_orth.negate();
443 
444  for ( DomainConstIterator it = aSet.domain().begin(),
445  it_end = aSet.domain().end(); it != it_end; ++it )
446  if ( hs( *it ) ) aSet.insert( *it );
447  for ( DigitalSetIterator it_set = aSet.begin(), it_set_end = aSet.end();
448  it_set != it_set_end; )
449  {
450  DigitalSetIterator it_cur = it_set; ++it_set;
451  if ( ! ( hs_orth( *it_cur )
452  && hs2( *it_cur )
453  && hs2_orth( *it_cur ) ) )
454  aSet.erase( it_cur );
455  }
456  return;
457  }
458  // case 4 : >= 3 vertices
459  ConstIterator it_vtx = begin();
460  ConstIterator it_vtx_end = end();
461  HalfSpace hs = halfSpace( it_vtx );
462  for ( DomainConstIterator it = aSet.domain().begin(),
463  it_end = aSet.domain().end(); it != it_end; ++it )
464  if ( hs( *it ) ) aSet.insert( *it );
465  ++it_vtx;
466  for ( ; it_vtx != it_vtx_end; ++it_vtx )
467  {
468  hs = halfSpace( it_vtx );
469  for ( DigitalSetIterator it_set = aSet.begin(), it_set_end = aSet.end();
470  it_set != it_set_end; )
471  {
472  DigitalSetIterator it_cur = it_set; ++it_set;
473  if ( ! hs( *it_cur ) ) aSet.erase( it_cur );
474  }
475  }
476 }

577  // _u, _v pair of Bezout vectors.

581  _B = outPts[0];
582 
583  // while A and B do not lie on the supporting line of hs2
584  // and satisfies hs3 and while v is not parallel to hs2
585  while ( ( ! hs2.isOnBoundary( _A ) ) // stops if A reaches hs2.
586  && ( hs3( _A ) ) // stops if A does not satisfy hs3.
587  && ( ! hs2.isOnBoundary( _B ) ) // stops if B reaches hs2.
588  && ( hs3( _B ) ) // stops if B does not satisfy hs3.

598  _B, _v, hs2.N, hs2.c );

600  if ( hs3( _u ) )
601  { // Point is still within bounds.

615  std::cerr << "[CIP::getAllPointsOfHull] add B=" << _B << std::endl;
616 #endif //ifdef DEBUG_ConvexIntegerPolygon
617  ASSERT( hs3( _B ) );
618  break; // necessarily the last point.
619  }
620  }
621  else
622  { //----- first configuration, we find a new A -----------------------
623  _ic.getCoefficientIntersection( _fl, _ce,
624  _A, _v, hs2.N, hs2.c );
625  _u = _A; _u += _v *_fl;
626  if ( hs3( _u ) )
627  { // Point is still within bounds.

642 #endif //ifdef DEBUG_ConvexIntegerPolygon

651  // JOL: to check.
652  _ic.getValidBezout( _v, _A, _u, hs2.N, hs2.c, hs2.N, hs2.c, false );
653  }
654  // when the loop finishes, we have to complete the computation
655  // of each convex hull
656  // if ( ! hs3( _A ) ) // A does not satisfy C3, we remove it.
657  // inPts.pop_back();
658  // else if ( ! hs3( _B ) ) // B does not satisfy C3, we remove it
659  // outPts.pop_back();
660  // else

662  outPts.push_back( _A );
663  else if ( hs2.isOnBoundary( _B ) ) // B lies on C2 so B is also a vertex of inPts
664  inPts.push_back( _B );
665 }

690 template <typename OutputIterator>
691 OutputIterator
692 DGtal::ConvexIntegerPolygon<TSpace,TSequence>::
693 computeConvexHullBorder
694 ( OutputIterator itOut,
695  const Point & pointRefC1,
696  const Point & pointRefC3,
697  const HalfSpace & hs1,
698  const HalfSpace & hs2,
699  const HalfSpace & hs3 ) const
700 {
701  // _u, _v: vectors u and v to determine the next vertex
702  bool exactIntersection;
703  // initializes A, B, u, v and the two first vertices of resultup and
704  // resultdown.
705  _inPts.resize( 1 ); //to store half convex hull border
706  _outPts.resize( 1 ); //to store half convex hull border
707  _inPts[ 0 ] = pointRefC1;
708 
709  // exactIntersection is equal to one when the intersection of the
710  // supporting lines of C1 and C2 corresponds to an integer point
711  exactIntersection = getFirstPointsOfHull( _v, _inPts[ 0 ], _outPts[ 0 ],
712  hs1, hs2 );
713  if ( ! exactIntersection ) // not integer intersection
714  {
715  //computation of the first part of the border
716  getAllPointsOfHull( _inPts, _outPts, _v, hs2, hs3 );
717  }
718  // fill in convexup
719  for( Size i = 0; i < _inPts.size(); ++i )
720  *itOut++ = _inPts[ i ];
721 
722  // second part
723  // initializes A, B, u, v and the two first vertices of resultup and
724  // resultdown.
725  _inPts.resize( 1 );
726  _outPts.resize( 1 );
727  _inPts[0] = pointRefC3;
728  // exactIntersection is equal to one when the intersection of the
729  // supporting lines of C3 and C2 corresponds to an integer point
730  exactIntersection = getFirstPointsOfHull( _v, _inPts[ 0 ], _outPts[ 0 ],
731  hs3, hs2 );
732  if ( ! exactIntersection ) // not integer intersection
733  {
734  //computation of the second part of the border
735  getAllPointsOfHull( _inPts, _outPts, _v, hs2, hs1 ); // check not hs1.
736  }
737  //fill in convexup
738  for( Size i = _inPts.size(); i != 0; --i )
739  {

741  }
