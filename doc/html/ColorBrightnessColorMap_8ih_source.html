177 template <typename PValue, int PDefaultColor>
178 inline
179 void
180 DGtal::ColorBrightnessColorMap<PValue,PDefaultColor>::HSVtoRGB
181 ( double &r, double &g, double &b,
182  const double h, const double s, const double v)
183 {
184  int i;
185  double f, p, q, t;
186  if( s == 0 ) { // achromatic (gray)
187  r = g = b = v;
188  return;
189  }
190  i = static_cast<int>( floor( h / 60 ) );
191  f = ( h / 60 ) - i; // factorial part of h
192  p = v * ( 1 - s );
193  q = v * ( 1 - s * f );
194  t = v * ( 1 - s * ( 1 - f ) );
195  switch( i ) {
196  case 0:
197  r = v; g = t; b = p;
198  break;
199  case 1:
200  r = q; g = v; b = p;
201  break;
202  case 2:
203  r = p; g = v; b = t;
204  break;
205  case 3:
206  r = p; g = q; b = v;
207  break;
208  case 4:
209  r = t; g = p; b = v;
210  break;
211  default: // case 5:
212  r = v; g = p; b = q;
213  break;
214  }
215 }

218 inline
219 void
220 DGtal::ColorBrightnessColorMap<PValue,PDefaultColor>::RGBtoHSV
221 ( double &h, double &s, double &v,
222  const unsigned char r,
223  const unsigned char g,
224  const unsigned char b )
225 {
226  double min = (r<g) ? r : g;
227  if ( b < min ) min = b;
228  unsigned char max = (r>g) ? r : g;
229  if ( b > max ) max = b;
230 
231  double dr = r / 255.0;
232  double dg = g / 255.0;
233  double db = b / 255.0;
234  v = max / 255.0; // (0.3*dr + 0.59*dg + 0.11*db);
235  if ( max == min ) {
236  h = 0;
237  s = 0;
238  return;
239  } else {
240  double diff = ( max - min ) / 255.0;
241  if ( max == r ) {
242  h = (dg - db ) / diff;
243  } else if ( max == g ) {
244  h = 2.0 + ( ( db - dr ) / diff );
245  } else if ( max == b ) {
246  h = 4.0 + ( ( dr - dg ) / diff );
247  }
248  h *= 60.0;
249  if ( h < 0 ) h += 360;
250  s = diff / v;
251  }
252 }
